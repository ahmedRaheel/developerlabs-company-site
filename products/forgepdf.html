Installation

Clone

bash
git clone https://github.com/your-repo/ForgePDF.git
cd ForgePDF

Build

bash
dotnet build

Run Sample API

bash
dotnet run --project samples/Enterprise.MinimalApi

Minimal API Example

csharp
using ForgePDF.Core;
using ForgePDF.Threading;
using ForgePDF.AspNetCore;

var builder = WebApplication.CreateBuilder(args);

builder.Services.AddForgePdf(options =>
{
    options.MaxConcurrentDocuments = Environment.ProcessorCount;
    options.QueueTimeout = TimeSpan.FromSeconds(30);
});

var app = builder.Build();

app.MapPost("/convert/html-to-pdf", async (
    HtmlToPdfRequest request,
    DocumentConversionService converter) =>
{
    var result = await converter.ConvertHtmlAsync(request);

    return Results.File(
        result.Bytes,
        "application/pdf",
        result.FileName);
});

app.Run();
